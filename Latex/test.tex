% Define the chapter start page style with color
\newcommand{\chapterstartpage}[3]{
    \newpage % Start a new page
    \backgroundsetup{
        scale=1,
        color=#3,
        opacity=1,
        angle=0,
        position=current page.north,
        vshift=-2cm,
        hshift=0cm,
        contents={\begin{tikzpicture}[remember picture,overlay]
          \fill[color=#3] (current page.north west) rectangle ([yshift=-4.78cm]current page.north east);
        \end{tikzpicture}}
      }
    \BgThispage % Apply the background to this page only
    \color{white} % Set text color to white
    \noindent\sffamily{\sbseries\fontsize{22}{26}\selectfont #1} % Chapter title in Source Sans SemiBold, size 22
    \newline\vspace{0pt} % Space between line and subtitle
    \newline\noindent\rule{\textwidth}{0.8pt} % White line 0.8pt thick
    \newline\vspace{4pt} % Space between line and subtitle
    \newline\sffamily{\bfseries\fontsize{7}{9}\selectfont\textls[300]{#2}} % Subtitle in bold, size 7, all caps, letter spacing 30%
    \newline\vspace{10pt} % Space after subtitle
    \color{black} % Set text color back to black
}
